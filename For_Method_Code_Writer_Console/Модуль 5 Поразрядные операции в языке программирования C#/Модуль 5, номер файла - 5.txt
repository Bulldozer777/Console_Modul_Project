 public static void E_5_Bitwise_Operations_Create_Logic_The_Shift()
        {
            Console.WriteLine("Операции сдвига");
            int x = 4;//100 двоичное
            Console.WriteLine(x << 1); //1000 - 8
            Console.WriteLine(x << 2); //10000-16
            Console.WriteLine(x >> 1); //10 - 2
            Console.WriteLine(x >> 2); //1 -1 
        }
